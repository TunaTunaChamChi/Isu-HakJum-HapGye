import asyncio import io import re import pandas as pd from js import document, Uint8Array, File, URL from pyodide.ffi import create_proxy def log(message): log_box = document.getElementById("status-log") log_box.innerText += f"\n{message}" log_box.scrollTop = log_box.scrollHeight log("시스템 준비 완료. 파일을 업로드해 주세요.") async def process_file(event): document.getElementById("loading-msg").style.display = "block" document.getElementById("download-area").style.display = "none" file_list = event.target.files if len(file_list) == 0: return file = file_list.item(0) document.getElementById("file-name").innerText = file.name log(f"파일 로드 완료: {file.name}") array_buffer = await file.arrayBuffer() uint8_array = Uint8Array.new(array_buffer) file_bytes = bytearray(uint8_array) try: log("데이터 분석 중...") if file.name.lower().endswith('.csv'): df = pd.read_csv(io.BytesIO(file_bytes), dtype=str, header=None, encoding='utf-8-sig', on_bad_lines='skip') else: df = pd.read_excel(io.BytesIO(file_bytes), dtype=str, header=None) extracted_data = [] current_grade = "" current_class = "" log("교과별/항목별 학점 추출 및 매칭 연산 시작...") for index, row in df.iterrows(): col_a = str(row[0]).strip() if pd.notna(row[0]) else "" # 1. 학년/반 구조 파악 if "학년" in col_a and "반" in col_a: match = re.search(r'(\d+)\s*학년\s*(\d+)\s*반', col_a) if match: current_grade = match.group(1) current_class = match.group(2) continue elif re.match(r'^\d+[-]\d+$', col_a): parts = col_a.split('-') current_grade = parts[0] current_class = parts[1] continue # 2. 학생 데이터 신규 등록 (A열이 순수 번호숫자인 경우) if col_a.isdigit(): num = col_a name = "" base_credit = 0.0 # 행 전체에서 학생 이름 탐색 for c_idx in range(1, len(row)): if pd.notna(row[c_idx]): val = str(row[c_idx]).strip() if not name and re.match(r'^[가-힣]{2,5}$', val) and val != "성명": name = val # 우측 끝에서 기본 일반교과 이수학점 합계 추출 for val in reversed(row.tolist()): if pd.notna(val): val_str = str(val).strip() if re.match(r'^\d+(\.\d+)?$', val_str) and val_str != num: try: base_credit = float(val_str) except: base_credit = 0.0 break extracted_data.append({ "학년": f"{current_grade}" if current_grade else "", "반": f"{current_class}" if current_class else "", "번호": f"{num}", "성명": name, "기본교과": base_credit, "특수교과": 0.0 }) # 3. 체육·예술 / 과학탐구실험 하단 합계 행 추적 및 누적 연산 elif col_a == "" and len(extracted_data) > 0: row_cells = [str(x).strip() for x in row.tolist() if pd.notna(x)] row_str = "".join(row_cells).replace(" ", "") # 하단에 분리된 특수과목 합계 행 식별 패턴 if "합계" in row_str or "이수학점" in row_str: # 해당 행의 가장 우측 숫자를 특수과목 학점으로 가져옴 for val in reversed(row.tolist()): if pd.notna(val): val_str = str(val).strip() if re.match(r'^\d+(\.\d+)?$', val_str): try: # 가장 최근에 추가된(현재 해당 구역의) 학생 데이터에 학점 합산 추가 extracted_data[-1]["`특수교과`"] += float(val_str) except: pass break # 4. 최종 결과 테이블 정비 및 수치 연산 결과 포맷팅 final_list = [] for s in extracted_data: total_sum = s["기본교과"] + s["`특수교과`"] # 소수점 아래가 0이면 정수로, 있으면 소수점 유지 표시 total_str = str(int(total_sum)) if total_sum.is_integer() else f"{total_sum:.1f}" final_list.append({ "학년": s["학년"], "반": s["반"], "번호": s["번호"], "성명": s["성명"], "이수학점 합계": total_str }) result_df = pd.DataFrame(final_list) if result_df.empty: log("⚠️ 데이터를 파싱하지 못했습니다. 원본 데이터 양식을 확인해주세요.") document.getElementById("loading-msg").style.display = "none" return result_df = result_df[["학년", "반", "번호", "성명", "이수학점 합계"]] # 5. 메모리 내 가상 엑셀 쓰기 output = io.BytesIO() with pd.ExcelWriter(output, engine='openpyxl') as writer: result_df.to_excel(writer, index=False, sheet_name="학점합산결과") output.seek(0) js_bytes = Uint8Array.new(output.read()) js_file = File.new([js_bytes], "extracted_total_credits.xlsx", { "type": "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet" }) file_url = URL.createObjectURL(js_file) download_link = document.getElementById("download-link") download_link.href = file_url document.getElementById("loading-msg").style.display = "none" document.getElementById("download-area").style.display = "block" log(f"정산 성공: 총 {len(result_df)}명의 이수학점 정밀 합산이 완료되었습니다.") except Exception as e: log(f"❌ 에러 발생: {str(e)}") document.getElementById("loading-msg").style.display = "none" file_input = document.getElementById("excel-file") event_proxy = create_proxy(process_file) file_input.addEventListener("change", event_proxy)
