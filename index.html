import asyncio import io import re import pandas as pd from js import document, Uint8Array, File, URL from pyodide.ffi import create_proxy def log(message): log_box = document.getElementById("status-log") log_box.innerText += f"\n{message}" log_box.scrollTop = log_box.scrollHeight log("시스템 준비 완료. 파일을 업로드해 주세요.") async def process_file(event): document.getElementById("loading-msg").style.display = "block" document.getElementById("download-area").style.display = "none" file_list = event.target.files if len(file_list) == 0: return file = file_list.item(0) document.getElementById("file-name").innerText = file.name log(f"파일 로드 완료: {file.name}") array_buffer = await file.arrayBuffer() uint8_array = Uint8Array.new(array_buffer) file_bytes = bytearray(uint8_array) try: log("데이터 구조 해석 중...") if file.name.lower().endswith('.csv'): df = pd.read_csv(io.BytesIO(file_bytes), dtype=str, header=None, encoding='utf-8-sig', on_bad_lines='skip') else: df = pd.read_excel(io.BytesIO(file_bytes), dtype=str, header=None) students_dict = {} current_grade = "" current_class = "" last_student_key = None log("학생별 데이터 정밀 그룹화 및 합산 진행 중...") for index, row in df.iterrows(): col_a = str(row[0]).strip() if pd.notna(row[0]) else "" # 1. 학년/반 파악 if "학년" in col_a and "반" in col_a: match = re.search(r'(\d+)\s*학년\s*(\d+)\s*반', col_a) if match: current_grade = match.group(1) current_class = match.group(2) continue elif re.match(r'^\d+[-]\d+$', col_a): parts = col_a.split('-') current_grade = parts[0] current_class = parts[1] continue # 행 전체 텍스트 병합 (검색 효율화) row_cells = [str(x).strip() for x in row.tolist() if pd.notna(x)] row_str = "".join(row_cells).replace(" ", "") if not row_str: continue # 2. 새로운 학생 등장 행 처리 (A열이 숫자) if col_a.isdigit(): num = col_a name = "" base_credit = 0.0 # 성명 검색 for c_idx in range(1, min(5, len(row))): if pd.notna(row[c_idx]): val = str(row[c_idx]).strip() if not name and re.match(r'^[가-힣]{2,5}$', val) and val != "성명": name = val # 해당 행의 이수학점 합계 검색 (우측 끝 정적 탐색 공정) for val in reversed(row.tolist()): if pd.notna(val): val_str = str(val).strip() if re.match(r'^\d+(\.\d+)?$', val_str) and val_str != num: try: base_credit = float(val_str) except: base_credit = 0.0 break # 학생 고유 키 설정 (중복 방지 및 1인 1행 보장) student_key = (current_grade, current_class, num, name) last_student_key = student_key if student_key not in students_dict: students_dict[student_key] = { "base": base_credit, "extra": 0.0 } else: # 중복 데이터 행이 또 들어와도 값 보존 처리 students_dict[student_key]["base"] = max(students_dict[student_key]["base"], base_credit) # 3. 체육·예술 / 과학탐구실험 하단 합계 행 정밀 타겟팅 elif col_a == "" and last_student_key is not None: # '교과이수학점합계' 문구와 '체육', '예술', '과학탐구' 등의 연관 단어가 있을 때만 수집 if "이수학점" in row_str or "합계" in row_str: for val in reversed(row.tolist()): if pd.notna(val): val_str = str(val).strip() if re.match(r'^\d+(\.\d+)?$', val_str): try: # 해당 학생의 추가 특수교과 학점 누적 (오타 교정 완료) students_dict[last_student_key]["extra"] += float(val_str) except: pass break # 4. 딕셔너리 데이터를 최종 엑셀 포맷 리스트로 전환 final_list = [] for (grade, clsw, num, name), credits in students_dict.items(): total_sum = credits["base"] + credits["extra"] # 예시 파일 형식처럼 소수점 뒤가 .0이면 정수로 표기 if total_sum.is_integer(): total_str = str(int(total_sum)) else: total_str = f"{total_sum:.1f}" final_list.append({ "학년": f"{grade}학년" if grade else "", "반": f"{clsw}반" if clsw else "", "번호": f"{num}번", "성명": name, "이수학점 합계": total_str }) result_df = pd.DataFrame(final_list) if result_df.empty: log("⚠️ 데이터를 추출하지 못했습니다. 파일 구조를 재확인해 주세요.") document.getElementById("loading-msg").style.display = "none" return # 컬럼 순서 지정 result_df = result_df[["학년", "반", "번호", "성명", "이수학점 합계"]] # 5. 다운로드용 파일 빌드 output = io.BytesIO() with pd.ExcelWriter(output, engine='openpyxl') as writer: result_df.to_excel(writer, index=False, sheet_name="학점추출결과") output.seek(0) js_bytes = Uint8Array.new(output.read()) js_file = File.new([js_bytes], "extracted_students_fixed.xlsx", { "type": "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet" }) file_url = URL.createObjectURL(js_file) download_link = document.getElementById("download-link") download_link.href = file_url document.getElementById("loading-msg").style.display = "none" document.getElementById("download-area").style.display = "block" log(f"정산 성공: 총 {len(result_df)}명의 중복 없는 합산 정산이 완료되었습니다.") except Exception as e: log(f"❌ 에러 발생: {str(e)}") document.getElementById("loading-msg").style.display = "none" file_input = document.getElementById("excel-file") event_proxy = create_proxy(process_file) file_input.addEventListener("change", event_proxy)
