import asyncio import io import re import pandas as pd from js import document, Uint8Array, File, URL from pyodide.ffi import create_proxy def log(message): log_box = document.getElementById("status-log") log_box.innerText += f"\n{message}" log_box.scrollTop = log_box.scrollHeight log("시스템 준비 완료. 파일을 업로드해 주세요.") async def process_file(event): document.getElementById("loading-msg").style.display = "block" document.getElementById("download-area").style.display = "none" file_list = event.target.files if len(file_list) == 0: return file = file_list.item(0) document.getElementById("file-name").innerText = file.name log(f"파일 로드 완료: {file.name}") array_buffer = await file.arrayBuffer() uint8_array = Uint8Array.new(array_buffer) file_bytes = bytearray(uint8_array) try: log("데이터 구조 분석 중...") if file.name.lower().endswith('.csv'): df = pd.read_csv(io.BytesIO(file_bytes), dtype=str, header=None, encoding='utf-8-sig', on_bad_lines='skip') else: df = pd.read_excel(io.BytesIO(file_bytes), dtype=str, header=None) students_dict = {} current_grade = "" current_class = "" last_student_key = None log("오직 [이수학점합계] 행의 G열 타겟팅 중...") for index, row in df.iterrows(): col_a = str(row[0]).strip() if pd.notna(row[0]) else "" # 1. 학년/반 파악 if "학년" in col_a and "반" in col_a: match = re.search(r'(\d+)\s*학년\s*(\d+)\s*반', col_a) if match: current_grade = match.group(1) current_class = match.group(2) continue elif re.match(r'^\d+[-]\d+$', col_a): parts = col_a.split('-') current_grade = parts[0] current_class = parts[1] continue row_cells = [str(x).strip() for x in row.tolist() if pd.notna(x)] row_str = "".join(row_cells).replace(" ", "") if not row_str: continue # 2. 학생 데이터 신규 등록 (A열이 숫자일 때) if col_a.isdigit(): num = col_a name = "" # 성명 탐색 (이 줄에서는 학점을 찾지 않습니다!) for c_idx in range(1, min(5, len(row))): if pd.notna(row[c_idx]): val = str(row[c_idx]).strip() if not name and re.match(r'^[가-힣]{2,5}$', val) and val != "성명": name = val student_key = (current_grade, current_class, num, name) last_student_key = student_key # 새 학생 등장 시 초기화 if student_key not in students_dict: students_dict[student_key] = 0.0 # 3. 오직 "이수학점합계" 글자가 포함된 행에서만 G열 값 누적 elif last_student_key is not None: if "이수학점합계" in row_str: if len(row) > 6: col_g = str(row[6]).strip() if pd.notna(row[6]) else "" if re.match(r'^\d+(\.\d+)?$', col_g): students_dict[last_student_key] += float(col_g) # 4. 결과 리스트 정돈 final_list = [] for (grade, clsw, num, name), total_sum in students_dict.items(): if total_sum.is_integer(): total_str = str(int(total_sum)) else: total_str = f"{total_sum:.1f}" final_list.append({ "학년": f"{grade}학년" if grade else "", "반": f"{clsw}반" if clsw else "", "번호": f"{num}번", "성명": name, "이수학점 합계": total_str }) result_df = pd.DataFrame(final_list) if result_df.empty: log("⚠️ 데이터를 정산하지 못했습니다. 파일 구조를 확인해 주세요.") document.getElementById("loading-msg").style.display = "none" return result_df = result_df[["학년", "반", "번호", "성명", "이수학점 합계"]] # 5. 가상 엑셀 출력 및 다운로드 연동 output = io.BytesIO() with pd.ExcelWriter(output, engine='openpyxl') as writer: result_df.to_excel(writer, index=False, sheet_name="학점추출결과") output.seek(0) js_bytes = Uint8Array.new(output.read()) js_file = File.new([js_bytes], "추출_학생이수학점.xlsx", { "type": "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet" }) file_url = URL.createObjectURL(js_file) download_link = document.getElementById("download-link") download_link.href = file_url document.getElementById("loading-msg").style.display = "none" document.getElementById("download-area").style.display = "block" log(f"정산 완료: 총 {len(result_df)}명의 학점이 G열 기준으로 완벽히 결산되었습니다.") except Exception as e: log(f"❌ 에러 발생: {str(e)}") document.getElementById("loading-msg").style.display = "none" file_input = document.getElementById("excel-file") event_proxy = create_proxy(process_file) file_input.addEventListener("change", event_proxy)
